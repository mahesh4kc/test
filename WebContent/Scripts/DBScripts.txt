drop table customer;

CustomerName varchar(50),
RelationShip char(1),

Address varchar(50),

PhoneNo bigint(15),
MobileNo bigint(15),
MailID varchar(50)
);

DROP TABLE BILL_DETAIL;
CREATE TABLE BILL_DETAIL
(
BILL_SEQUENCE INT ,
PRODUCT_NO INT ,
PRODUCT_DESCRIPTION VARCHAR(2000) NOT NULL ,

PRIMARY KEY (BILL_SEQUENCE,PRODUCT_NO)
);
DROP TABLE BILL_HEADER;
CREATE TABLE BILL_HEADER(
BILL_SEQUENCE INT PRIMARY KEY,
BILL_SERIAL CHAR(1) NOT NULL,
BILL_NO INT NOT NULL, 
BILL_DATE varchar(12),
CUSTOMERID INT NOT NULL,
CAREOF VARCHAR(500),
PRODUCT_TYPE_NO INT  NOT NULL,
RATE_OF_INTEREST DECIMAL(6,3)  NOT NULL,
AMOUNT INT  NOT NULL,
AMOUNT_IN_WORDS varchar(200)  NOT NULL,
PRESENT_VALUE INT  NOT NULL,

MONTHLY_INCOME INT  NOT NULL DEFAULT 8600,
REDEMPTION_DATE  varchar(12),
REDEMPTION_INTEREST DECIMAL(10,3),
REDEMPTION_TOTAL DECIMAL(20,3),
REDEMPTION_STATUS  CHAR(1) DEFAULT 'O'
);
